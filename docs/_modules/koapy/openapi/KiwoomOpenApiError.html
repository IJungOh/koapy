        if callable(code_or_function):
            code = code_or_function(*args, **kwargs)
        elif isinstance(code_or_function, int):
            code = code_or_function
        else:
            raise ValueError("Invalid argument for code_or_function: %s" % code_or_function)
        if code < 0:
            raise cls(code)
        return code
