    def __init__(self, queue=None):
        if queue is None:

        self._queue = queue
        self._sentinel = object()

[docs]    def __iter__(self):
        while True:
            try:
                value = self._queue.get(True, self._queue_get_timeout)
            except Empty:
                pass
            else:
                if value == self._sentinel:
                    self._queue.task_done()
                    break
                if isinstance(value, Exception):
                    self._queue.task_done()
                    raise value
                yield value
                self._queue.task_done()
